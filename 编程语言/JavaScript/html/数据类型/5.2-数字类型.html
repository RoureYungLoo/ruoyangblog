// toString(base) 返回base进制的字符串表示,base 的范围可以从 2 到 36。默认情况下是 10。

            



            
let num = 255

            
let bignum = 12345678

console.log(num.toString(8)) // 377

console.log(Math.round(PI))

console.log(+PI.toFixed(8))
